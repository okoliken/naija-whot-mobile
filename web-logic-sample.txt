import { computed, ref } from 'vue'
import { defineStore } from 'pinia'
import { pickComputerMove } from '../services/aiPlayer'
import { playSfx, unlockAudio } from '../services/sounds'
import { loadStats, recordResult } from '../services/stats'
import { isMarketDepletedError, WhotEngine } from '../services/whotEngine'
import { DIFFICULTIES } from '../types/game'
import type {
  CardModel,
  CardShape,
  Difficulty,
  GamePhase,
  StatsSnapshot,
  TurnOwner,
} from '../types/game'

type PendingPenalty = 2 | 5 | null

export const useGameStore = defineStore('game', () => {
  const engine = ref<WhotEngine | null>(null)
  const phase = ref<GamePhase>('idle')
  const activeTurn = ref<TurnOwner>('human')
  const humanHand = ref<CardModel[]>([])
  const computerHand = ref<CardModel[]>([])
  const topCard = ref<CardModel | null>(null)
  const requestedShape = ref<CardShape | null>(null)
  const pendingPickCount = ref(0)
  const pendingPenalty = ref<PendingPenalty>(null)
  const pendingSkipCount = ref(0)
  const lastEventMessage = ref('Start a new game to play.')
  const winner = ref<TurnOwner | null>(null)
  const stats = ref<StatsSnapshot>(loadStats())
  const muted = ref(false)
  const selectingShape = ref(false)
  const pendingWhotIndex = ref<number | null>(null)
  const theme = ref<'dark' | 'light'>('dark')
  const feedOpen = ref(false)
  const resetConfirmOpen = ref(false)
  const roundEndOpen = ref(false)
  const difficultyOpen = ref(false)
  const difficulty = ref<Difficulty>('medium')
  const eventLog = ref<{ id: number; message: string; at: number }[]>([])
  let eventCounter = 0

  const marketCount = computed(() => engine.value?.getMarketCount() ?? 0)
  const hasPendingShapeChoice = computed(() => pendingWhotIndex.value !== null)
  const isGameInProgress = computed(
    () => phase.value !== 'idle' && phase.value !== 'roundEnd' && !!engine.value,
  )

  function pushEvent(message: string) {
    eventCounter += 1
    eventLog.value.unshift({ id: eventCounter, message, at: Date.now() })
    if (eventLog.value.length > 50) eventLog.value.length = 50
    lastEventMessage.value = message
  }

  const canPlayAt = (index: number) => {
    const card = humanHand.value[index]
    if (!card || !topCard.value || !engine.value || activeTurn.value !== 'human') return false
    if (pendingPenalty.value === 2 && pendingPickCount.value > 0) return card.value === 2
    if (pendingPenalty.value === 5 && pendingPickCount.value > 0) return card.value === 5
    return engine.value.canPlayCard(card, topCard.value, requestedShape.value)
  }

  function startGame() {
    engine.value = new WhotEngine()
    activeTurn.value = 'human'
    phase.value = 'playerTurn'
    requestedShape.value = null
    pendingPickCount.value = 0
    pendingPenalty.value = null
    pendingSkipCount.value = 0
    winner.value = null
    selectingShape.value = false
    pendingWhotIndex.value = null
    syncHands()
    topCard.value = engine.value.getTopCard()
    eventLog.value = []
    pushEvent('Game started. Your turn.')
  }

  function initializePreferences() {
    const savedTheme = localStorage.getItem('naija-whot.theme')
    if (savedTheme === 'light' || savedTheme === 'dark') {
      theme.value = savedTheme
    }
    document.documentElement.setAttribute('data-theme', theme.value)

    const savedDifficulty = localStorage.getItem('naija-whot.difficulty')
    if (savedDifficulty === 'easy' || savedDifficulty === 'medium' || savedDifficulty === 'hard') {
      difficulty.value = savedDifficulty
    }
  }

  function openDifficulty() {
    difficultyOpen.value = true
  }

  function closeDifficulty() {
    difficultyOpen.value = false
  }

  function setDifficulty(next: Difficulty, restart = false) {
    difficulty.value = next
    localStorage.setItem('naija-whot.difficulty', next)
    pushEvent(`Difficulty set to ${next}.`)
    if (restart) {
      difficultyOpen.value = false
      startGame()
    }
  }

  function toggleTheme() {
    theme.value = theme.value === 'dark' ? 'light' : 'dark'
    localStorage.setItem('naija-whot.theme', theme.value)
    document.documentElement.setAttribute('data-theme', theme.value)
  }

  function syncHands() {
    if (!engine.value) return
    humanHand.value = engine.value.getHand(0)
    computerHand.value = engine.value.getHand(1)
  }

  function playHumanCard(index: number) {
    if (!engine.value || !topCard.value || phase.value === 'roundEnd') return
    if (!canPlayAt(index)) return

    const selectedCard = humanHand.value[index]
    if (!selectedCard) return

    if (selectedCard.value === 20) {
      pendingWhotIndex.value = index
      selectingShape.value = true
      phase.value = 'resolvingAction'
      pushEvent('Choose a shape for Crown (20).')
      return
    }

    let played: CardModel
    try {
      played = engine.value.play(0, index)
    } catch (err) {
      if (isMarketDepletedError(err)) {
        syncHands()
        pushEvent('Market dry while resolving last card.')
        resolveByExhaustion()
        return
      }
      throw err
    }
    topCard.value = played
    requestedShape.value = null
    syncHands()
    handleCardEffect(played, 'human')
    playSfx('play', muted.value)
    if (checkWinner()) return

    maybeSwitchAfterPlay(played, 'human')
  }

  function chooseShape(shape: CardShape) {
    if (!engine.value || phase.value === 'roundEnd') return
    requestedShape.value = shape
    selectingShape.value = false
    const selectedIndex = pendingWhotIndex.value
    pendingWhotIndex.value = null

    if (selectedIndex === null) {
      pushEvent(`Shape requested: ${shape}`)
      if (activeTurn.value === 'human') {
        activeTurn.value = 'computer'
        engine.value.switchTurn()
        phase.value = 'aiTurn'
        runComputerTurn()
      } else {
        activeTurn.value = 'human'
        engine.value.switchTurn()
        phase.value = 'playerTurn'
      }
      return
    }

    let played: CardModel
    try {
      played = engine.value.play(0, selectedIndex, shape)
    } catch (err) {
      if (isMarketDepletedError(err)) {
        syncHands()
        pushEvent('Market dry while resolving last card.')
        resolveByExhaustion()
        return
      }
      throw err
    }
    topCard.value = played
    syncHands()
    playSfx('play', muted.value)
    pushEvent(`You played Crown and requested ${shape}.`)
    if (checkWinner()) return
    activeTurn.value = 'computer'
    engine.value.switchTurn()
    phase.value = 'aiTurn'
    runComputerTurn()
  }

  function closeShapePrompt() {
    if (pendingWhotIndex.value !== null && phase.value !== 'roundEnd') {
      selectingShape.value = false
      pushEvent('Shape chooser closed. Resume to continue your Crown move.')
    }
  }

  function reopenShapePrompt() {
    if (pendingWhotIndex.value !== null && phase.value !== 'roundEnd') {
      selectingShape.value = true
    }
  }

  function drawHumanCard() {
    if (!engine.value || activeTurn.value !== 'human' || phase.value === 'roundEnd') return
    const drawCount = pendingPickCount.value > 0 ? pendingPickCount.value : 1
    let drawn = 0
    for (let i = 0; i < drawCount; i += 1) {
      const result = engine.value.draw(0)
      if (!result.ok) {
        syncHands()
        if (drawn > 0) playSfx('draw', muted.value)
        pushEvent(
          drawn > 0
            ? `Market ran dry after ${drawn} ${drawn === 1 ? 'card' : 'cards'}.`
            : 'Market is empty — round resolved by hand count.',
        )
        resolveByExhaustion()
        return
      }
      drawn += 1
    }
    playSfx('draw', muted.value)
    syncHands()
    pushEvent(drawCount > 1 ? `You picked ${drawCount} cards.` : 'You picked from market.')
    pendingPickCount.value = 0
    pendingPenalty.value = null
    activeTurn.value = 'computer'
    engine.value.switchTurn()
    phase.value = 'aiTurn'
    runComputerTurn()
  }

  function runComputerTurn() {
    if (
      !engine.value ||
      !topCard.value ||
      phase.value !== 'aiTurn' ||
      activeTurn.value !== 'computer'
    ) {
      return
    }
    const thinkMs = DIFFICULTIES.find((d) => d.id === difficulty.value)?.thinkingMs ?? 600
    window.setTimeout(() => {
      if (
        !engine.value ||
        !topCard.value ||
        phase.value !== 'aiTurn' ||
        activeTurn.value !== 'computer'
      ) {
        return
      }
      if (pendingSkipCount.value > 0) {
        pendingSkipCount.value -= 1
        activeTurn.value = 'human'
        engine.value.switchTurn()
        phase.value = 'playerTurn'
        pushEvent('Computer was suspended.')
        return
      }

      const decision = pickComputerMove(
        {
          hand: computerHand.value,
          topCard: topCard.value,
          requestedShape: requestedShape.value,
          opponentHandSize: humanHand.value.length,
          pendingPickCount: pendingPickCount.value,
          difficulty: difficulty.value,
        },
        (card, top, needed) => {
          if (!engine.value) return false
          if (pendingPenalty.value === 2 && pendingPickCount.value > 0) return card.value === 2
          if (pendingPenalty.value === 5 && pendingPickCount.value > 0) return card.value === 5
          return engine.value.canPlayCard(card, top, needed)
        },
      )

      if (decision.type === 'draw' || decision.cardIndex === undefined) {
        const drawCount = pendingPickCount.value > 0 ? pendingPickCount.value : 1
        let drawn = 0
        for (let i = 0; i < drawCount; i += 1) {
          const result = engine.value.draw(1)
          if (!result.ok) {
            syncHands()
            if (drawn > 0) playSfx('draw', muted.value)
            pushEvent(
              drawn > 0
                ? `CPU pulled ${drawn} before market dried up.`
                : 'Market is empty — round resolved by hand count.',
            )
            resolveByExhaustion()
            return
          }
          drawn += 1
        }
        pendingPickCount.value = 0
        pendingPenalty.value = null
        syncHands()
        playSfx('draw', muted.value)
        activeTurn.value = 'human'
        engine.value.switchTurn()
        phase.value = 'playerTurn'
        pushEvent(
          drawCount > 1 ? `Computer picked ${drawCount} cards.` : 'Computer picked from market.',
        )
        return
      }

      let played: CardModel
      try {
        played = engine.value.play(1, decision.cardIndex, decision.requestedShape)
      } catch (err) {
        if (isMarketDepletedError(err)) {
          syncHands()
          pushEvent('Market dry while CPU resolved last card.')
          resolveByExhaustion()
          return
        }
        throw err
      }
      topCard.value = played
      requestedShape.value = decision.requestedShape ?? null
      syncHands()
      handleCardEffect(played, 'computer')
      playSfx('play', muted.value)
      pushEvent(`Computer played ${played.shape} ${played.value}.`)
      if (checkWinner()) return

      maybeSwitchAfterPlay(played, 'computer')
    }, thinkMs)
  }

  function maybeSwitchAfterPlay(card: CardModel, player: TurnOwner) {
    if (phase.value === 'roundEnd') return
    const isHoldOn = card.value === 1
    if (isHoldOn) {
      activeTurn.value = player
      phase.value = player === 'human' ? 'playerTurn' : 'aiTurn'
      if (player === 'computer') runComputerTurn()
      return
    }

    if (player === 'human') {
      activeTurn.value = 'computer'
      engine.value?.switchTurn()
      phase.value = 'aiTurn'
      runComputerTurn()
    } else {
      activeTurn.value = 'human'
      engine.value?.switchTurn()
      phase.value = 'playerTurn'
    }
  }

  function handleCardEffect(card: CardModel, player: TurnOwner) {
    if (card.value === 2) {
      pendingPickCount.value += 2
      pendingPenalty.value = 2
      pushEvent(`${player === 'human' ? 'You' : 'Computer'} played Pick Two.`)
      return
    }
    if (card.value === 5) {
      pendingPickCount.value += 3
      pendingPenalty.value = 5
      pushEvent(`${player === 'human' ? 'You' : 'Computer'} played Pick Three.`)
      return
    }
    if (card.value === 8) {
      pendingSkipCount.value += 1
      pushEvent(`${player === 'human' ? 'You' : 'Computer'} played Suspension.`)
      return
    }
    if (card.value === 14 && engine.value) {
      const affected = player === 'human' ? 1 : 0
      const result = engine.value.draw(affected)
      syncHands()
      if (!result.ok) {
        pushEvent('General Market called, but the market is empty.')
        resolveByExhaustion()
        return
      }
      pushEvent(`${player === 'human' ? 'You' : 'Computer'} played General Market.`)
    }
  }

  function checkWinner() {
    if (humanHand.value.length === 0) {
      endRound('human')
      return true
    }
    if (computerHand.value.length === 0) {
      endRound('computer')
      return true
    }
    return false
  }

  function resolveByExhaustion() {
    const humanCount = humanHand.value.length
    const computerCount = computerHand.value.length
    let roundWinner: TurnOwner
    if (humanCount === computerCount) {
      const humanPoints = humanHand.value.reduce((sum, c) => sum + c.value, 0)
      const computerPoints = computerHand.value.reduce((sum, c) => sum + c.value, 0)
      roundWinner = humanPoints <= computerPoints ? 'human' : 'computer'
    } else {
      roundWinner = humanCount < computerCount ? 'human' : 'computer'
    }
    pushEvent(
      roundWinner === 'human'
        ? `Market dry — you win with ${humanCount} vs ${computerCount} cards.`
        : `Market dry — CPU wins with ${computerCount} vs ${humanCount} cards.`,
    )
    endRound(roundWinner)
  }

  function endRound(roundWinner: TurnOwner) {
    winner.value = roundWinner
    phase.value = 'roundEnd'
    stats.value = recordResult(stats.value, roundWinner)
    playSfx(roundWinner === 'human' ? 'win' : 'lose', muted.value)
    pushEvent(roundWinner === 'human' ? 'You won this round!' : 'Computer wins this round.')
    roundEndOpen.value = true
  }

  function closeRoundEnd() {
    roundEndOpen.value = false
  }

  function openRoundEnd() {
    if (winner.value) roundEndOpen.value = true
  }

  function openFeed() {
    feedOpen.value = true
  }

  function closeFeed() {
    feedOpen.value = false
  }

  function requestReset() {
    if (isGameInProgress.value) {
      resetConfirmOpen.value = true
    } else {
      startGame()
    }
  }

  function cancelReset() {
    resetConfirmOpen.value = false
  }

  function confirmReset() {
    resetConfirmOpen.value = false
    roundEndOpen.value = false
    startGame()
  }

  function toggleMute() {
    muted.value = !muted.value
  }

  async function unlockSound() {
    await unlockAudio()
  }

  return {
    phase,
    activeTurn,
    humanHand,
    computerHand,
    topCard,
    requestedShape,
    pendingPickCount,
    pendingSkipCount,
    lastEventMessage,
    marketCount,
    winner,
    stats,
    muted,
    theme,
    selectingShape,
    hasPendingShapeChoice,
    feedOpen,
    resetConfirmOpen,
    roundEndOpen,
    difficultyOpen,
    difficulty,
    eventLog,
    isGameInProgress,
    initializePreferences,
    startGame,
    playHumanCard,
    drawHumanCard,
    canPlayAt,
    chooseShape,
    closeShapePrompt,
    reopenShapePrompt,
    toggleMute,
    toggleTheme,
    unlockSound,
    runComputerTurn,
    openFeed,
    closeFeed,
    requestReset,
    cancelReset,
    confirmReset,
    closeRoundEnd,
    openRoundEnd,
    openDifficulty,
    closeDifficulty,
    setDifficulty,
  }
})
